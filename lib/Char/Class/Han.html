EXAMPLE

  use utf8;
  use 5.7.3;
  use Char::InSet::Han qw(InHanJP_Jimmei InHanJP_JimmeiKyoyou);
  $s = q(琉球諸島);
  ## Marks han characters included in 人名用漢字表・人名用漢字許容字体表
  $s =~ s/(\p{InHanJP_Jimmei}|\p{InHanJP_JimmeiKyoyou})/*$1/g;
  print $s;     ## *琉球諸島
